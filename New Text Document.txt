import com.networknt.schema.JsonSchemaFactory;
import com.networknt.schema.JsonValidator;
import com.networknt.schema.ValidationResult;

import java.io.File;
import java.io.FileInputStream;
import java.io.InputStream;

public class JsonValidationExample {
    public static void main(String[] args) {
        try {
            // Load the JSON schema file
            File schemaFile = new File("path/to/schema.json");
            InputStream schemaStream = new FileInputStream(schemaFile);

            // Load the JSON sample data
            File jsonDataFile = new File("path/to/sample.json");
            InputStream jsonDataStream = new FileInputStream(jsonDataFile);

            // Create a JsonSchemaFactory instance
            JsonSchemaFactory schemaFactory = JsonSchemaFactory.getInstance();

            // Compile the schema into a JsonValidator
            JsonValidator validator = schemaFactory.getSchema(schemaStream);

            // Validate the JSON sample
            ValidationResult validationResult = validator.validate(jsonDataStream);

            // Check validation results
            if (validationResult.isValid()) {
                System.out.println("JSON sample is valid.");
            } else {
                System.out.println("JSON sample is invalid.");
                System.out.println(validationResult.getErrors());
            }
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
